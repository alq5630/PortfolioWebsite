C Plus Plus Example

A basic linked list implementation demonstrating memory management and pointer navigation.


// Simple singly linked list example

struct Node {
    int value;
    Node* next;
};

class LinkedList {
private:
    Node* head;

public:
    LinkedList() : head(nullptr) {}

    void insertFront(int v) {
        Node* n = new Node;
        n->value = v;
        n->next = head;
        head = n;
    }

    int size() {
        int count = 0;
        Node* current = head;
        while (current != nullptr) {
            count++;
            current = current->next;
        }
        return count;
    }

    ~LinkedList() {
        Node* current = head;
        while (current != nullptr) {
            Node* temp = current->next;
            delete current;
            current = temp;
        }
    }
};

Java Example

A Java class demonstrating validation, object behavior, and basic method logic.


// Simple Java class example

public class Student {
    private String name;
    private int grade;

    public Student(String name, int grade) {
        if (grade < 0 || grade > 100) {
            throw new IllegalArgumentException("Grade must be between 0 and 100");
        }
        this.name = name;
        this.grade = grade;
    }

    public boolean isPassing() {
        return grade >= 70;
    }

    public String summary() {
        return name + " is " + (isPassing() ? "passing" : "not passing");
    }
}
